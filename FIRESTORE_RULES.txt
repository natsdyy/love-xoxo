// FIRESTORE SECURITY RULES
// Copy and paste these rules into Firebase Console > Firestore Database > Rules tab

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection - authenticated users only
    // Stores login records and user profiles
    match /users/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Stocks collection - authenticated users only
    // Stores all inventory items (subscriptions, services, products)
    match /stocks/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Sales collection - authenticated users only
    // Stores all sold transactions and revenue records
    match /sales/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Pending collection - authenticated users only
    // Stores reserved/pending transactions waiting for confirmation
    match /pending/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Refunds collection - authenticated users only
    // Stores refund records and refund transactions
    match /refunds/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Capital collection - authenticated users only
    // Stores cost per service/category for profit tracking
    match /capital/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Prices collection - authenticated users only
    // Stores selling prices per service/duration/category
    match /prices/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Discounts collection - authenticated users only
    // Stores discount rates and promotional pricing
    match /discounts/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Orders collection - authenticated users only
    match /orders/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Monitoring collection - authenticated users only
    match /monitoring/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Replacements collection - authenticated users only
    match /replacements/{document=**} {
      allow read, write: if request.auth != null;
    }

  }
}

// ============================================
// COLLECTIONS REFERENCE
// ============================================
// - users: Login records and user profiles (admins, owners)
// - stocks: Inventory items (subscriptions, services, products) - status: available/sold
// - sales: Completed sales transactions
// - pending: Reserved/pending transactions
// - refunds: Refund records and transactions
// - capital: Cost per service for profit calculations
// - prices: Selling prices per service/duration/category
// - discounts: Discount rates and promotional pricing
// - orders: Supplier orders and purchase orders
// - monitoring: Dynamic account slot tracking
// - replacements: Account fix/replacement history
